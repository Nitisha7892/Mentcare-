async function registerUser() { const name = document.getElementById("signupName").value; const email = document.getElementById("signupEmail").value; const password = document.getElementById("signupPassword").value; const res = await fetch("http://127.0.0.1:8000/api/register/", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ name, email, password }) }); const data = await res.json(); alert(data.message); }
