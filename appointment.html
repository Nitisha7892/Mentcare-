async function bookAppointment() { const name = document.getElementById("patientName").value; const email = document.getElementById("patientEmail").value; const date = document.getElementById("date").value; const time = document.getElementById("time").value; const doctor = document.getElementById("doctor").value; const res = await fetch("http://127.0.0.1:8000/api/appointment/", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ name, email, date, time, doctor }) }); const data = await res.json(); alert(data.message); }
